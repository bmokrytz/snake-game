269void initializeGameGrid() {
270 int rows = GAMEGRIDROWS, cols = GAMEGRIDCOLS;
271 gameBoard.grid = (GameBoardGrid)malloc(sizeof(GridCell*) * rows);
272 for (int i = 0; i < rows; i++) {
273 gameBoard.grid[i] = (GridCell*)malloc(sizeof(GridCell) * cols);
274 for (int j = 0; j < cols; j++) {
275 gameBoard.grid[i][j].x = i; gameBoard.grid[i][j].y = j;
276 gameBoard.grid[i][j].containsHead = 0;
277 gameBoard.grid[i][j].containsSnake = 0;
278 gameBoard.grid[i][j].containsFruit = 0;
279 if (i == 0 || j == 0 || i == (rows - 1) || j == (cols - 1)) gameBoard.grid[i][j].containsWall = 1;
280 else gameBoard.grid[i][j].containsWall = 0;
281 }
282 }
283 gameBoard.grid_rows = rows;
284 gameBoard.grid_cols = cols;
285}

522void moveSnake(void) {
523 if (snake.node == NULL) {
524 logError(L"Error in function moveSnake() of game.h.\n\tsnake.node == NULL\n");
525 }
526
527 SnakeNode* ptr = snake.node;
528 ptr->prev_x = ptr->x;
529 ptr->prev_y = ptr->y;
530 gameBoard.grid[ptr->x][ptr->y].containsHead = 0;
531
532 if (snake.movement_direction == DIRECTION_UP) {
533 ptr->y -= 1;
534 } else if (snake.movement_direction == DIRECTION_DOWN) {
535 ptr->y += 1;
536 } else if (snake.movement_direction == DIRECTION_LEFT) {
537 ptr->x -= 1;
538 } else if (snake.movement_direction == DIRECTION_RIGHT) {
539 ptr->x += 1;
540 }
541
542 gameBoard.grid[ptr->x][ptr->y].containsHead = 1;
543 ptr = ptr->nextNode;
544
545 while (ptr != NULL) {
546 ptr->prev_x = ptr->x;
547 ptr->prev_y = ptr->y;
548 gameBoard.grid[ptr->x][ptr->y].containsSnake = 0;
549
550 ptr->x = ptr->prevNode->prev_x;
551 ptr->y = ptr->prevNode->prev_y;
552
553 gameBoard.grid[ptr->x][ptr->y].containsSnake = 1;
554 ptr = ptr->nextNode;
555 }

602int collisionCheck() {
603 if (snake.node == NULL) {
604 logError(L"Error in function collisionCheck() of game.h.\n\tsnake.node == NULL\n");
605 }
606 if (gameBoard.grid[snake.node->x][snake.node->y].containsWall || gameBoard.grid[snake.node->x][snake.node->y].containsSnake) {
607 return COLLISION;
608 }
609 else if (gameBoard.grid[snake.node->x][snake.node->y].containsFruit) {

612 return NO_COLLISION;

678void updateGameboard(RECT mainWindowRect) {
679 int mainWindowWidth = mainWindowRect.right - mainWindowRect.left;
680 int mainWindowHeight = mainWindowRect.bottom - mainWindowRect.top;
681 gameBoard.rect.height = GAMEBOARDHEIGHT; gameBoard.rect.width = GAMEBOARDWIDTH;
682 gameBoard.rect.left = (mainWindowWidth - gameBoard.rect.width) / 2;
683 gameBoard.rect.top = (mainWindowHeight - gameBoard.rect.height) / 2;
684 gameBoard.rect.right = gameBoard.rect.left + gameBoard.rect.width;
685 gameBoard.rect.bottom = gameBoard.rect.top + gameBoard.rect.height;
686 if ((GAMEBOARDWIDTH % GAMEGRIDCOLS || GAMEBOARDHEIGHT % GAMEGRIDROWS) != 0) {
687 logError(L"Error in function setupGridCellDimensions() of game.h.\n\t(gameBoard.rect.width \% gameBoard.grid_cols) != 0\n");
688 }
689 gameBoard.cell_width = (gameBoard.rect.width / gameBoard.grid_cols);
690 gameBoard.cell_height = (gameBoard.rect.height / gameBoard.grid_rows);

339LRESULT CALLBACK SnakeWindowProc(HWND hwnd, UINT uMsg, WPARAM wParam, LPARAM lParam)
340{
341 switch (uMsg)
342 {
343 case WM_DESTROY:
344 {
345 windowCleanUp();
346 freeGameData();
347 PostQuitMessage(0);
348 return 0;
349 }
350 case WM_SETCURSOR:
351 {
352 DefWindowProc(hwnd, uMsg, wParam, lParam);
353 return 0;
354 }
355
356 case WM_PAINT:
357 {
358 if (hwnd == mainWindow) {
359 paintMainWindow();
360 }
361 else {
362 /*
363 RECT windowRect; GetClientRect(hwnd, &windowRect);
364 PAINTSTRUCT ps;
365 HDC hdc = BeginPaint(hwnd, &ps);
366 FillRect(hdc, &ps.rcPaint, fruitBrush);
367 */
368 paintGameWindow();
369 }
370 return 0;
371 }
372 case WM_KEYDOWN:
373 {
374
375 if (wParam == 'W' || wParam == 'w' || wParam == VK_UP) {
376 if (gameStatus != PAUSE_GAME) {
377 changeSnakeDirection(DIRECTION_UP);
378 }
379 }
380 else if (wParam == 'A' || wParam == 'a' || wParam == VK_LEFT) {
381 if (gameStatus != PAUSE_GAME) {
382 changeSnakeDirection(DIRECTION_LEFT);
383 }
384 }
385 else if (wParam == 'S' || wParam == 's' || wParam == VK_DOWN) {
386 if (gameStatus != PAUSE_GAME) {
387 changeSnakeDirection(DIRECTION_DOWN);
388 }
389 }
390 else if (wParam == 'D' || wParam == 'd' || wParam == VK_RIGHT) {
391 if (gameStatus != PAUSE_GAME) {
392 changeSnakeDirection(DIRECTION_RIGHT);
393 }
394 }
395 else if (wParam == VK_RETURN) {
396 /*
397 wchar_t errMsg[256];
398 if (hwnd == mainWindow) {
399 swprintf(errMsg, 256, L"mainWindow window received WM_KEYDOWN signal.\n");
400 }
401 if (hwnd == gameWindow) {
402 swprintf(errMsg, 256, L"gameWindow window received WM_KEYDOWN signal.\n");
403 }
404 logDebugMessage(errMsg);
405 */
406 togglePause();
407 }
408 return 0;
409 }
410 case WM_TIMER:
411 {
412 if (gameStatus == START_GAME) {
413 generateNextFrame();
414 RECT gameField; GetClientRect(gameWindow, &gameField);
415 gameField.left = gameBoard.cell_width;
416 gameField.right = gameField.right - gameBoard.cell_width;
417 gameField.top = gameBoard.cell_height;
418 gameField.bottom = gameField.bottom - gameBoard.cell_height;
419 InvalidateRect(gameWindow, &gameField, TRUE);
420 }
421 return 0;
422 }
423 case WM_SIZE:
424 {
425 updateGameboardPos();
426 GameBoardRect gameboardRect = getGameboardRect();
427 MoveWindow(gameWindow, gameboardRect.left, gameboardRect.top, gameboardRect.width, gameboardRect.height, TRUE);
428 InvalidateRect(mainWindow, NULL, TRUE);
429 InvalidateRect(gameWindow, NULL, TRUE);
430 return 0;
431 }
432 }
433 return DefWindowProc(hwnd, uMsg, wParam, lParam);
434}

501void paintMainWindow() {
502 PAINTSTRUCT ps;
503 HDC hdc = BeginPaint(mainWindow, &ps);
504 FillRect(hdc, &ps.rcPaint, backgroundBrush);
505
506 const wchar_t *title = L"Snake Game";
507
508 GameBoardRect gameBoard = getGameboardRect();
509
510 RECT textRect = {
511 gameBoard.left,
512 gameBoard.top - 40, // 40 px above square
513 gameBoard.right,
514 gameBoard.top
515 };
516
517 SetBkMode(hdc, TRANSPARENT); // dont paint background behind text
518 SetTextColor(hdc, RGB(0, 0, 0)); // black text
519
520 DrawText(hdc, title, -1, &textRect, DT_CENTER | DT_VCENTER | DT_SINGLELINE);
521
522
523 EndPaint(mainWindow, &ps);

573void drawSnake(HDC hdc) {
574 if (snake.node == NULL) {
575 logError(L"Error in function drawSnake(HDC hdc) of window.h.\n\tsnake.node == NULL\n");
576 }
577 SnakeNode* node = snake.node;
578 HBRUSH oldBrush = (HBRUSH)SelectObject(hdc, snakeBrush);
579 while (node != NULL) {
580 RECT cell_bounds = getCellBoundingRect(node->x, node->y);
581 drawCircle(hdc, cell_bounds);
582 node = node->nextNode;
583 }
584 SelectObject(hdc, oldBrush);
585}

661void drawGridDebug(RECT field, HDC hdc) {
662 HPEN hPen = CreatePen(PS_SOLID, 1, RGB(0, 0, 0));
663 HPEN hOldPen = (HPEN)SelectObject(hdc, hPen);
664
665 int cell_width = getGameBoardCellWidth();
666 int cell_height = getGameBoardCellHeight();
667
668 for (int i = 1; i <= GAMEGRIDCOLS; i++) {
669 int col = field.left + (i * cell_width);
670 MoveToEx(hdc, col, field.top, NULL);
671 LineTo(hdc, col, field.bottom);
672 }
673
674 for (int i = 1; i <= GAMEGRIDROWS; i++) {
675 int row = field.top + (i * cell_height);
676 MoveToEx(hdc, field.left, row, NULL);
677 LineTo(hdc, field.right, row);
678 }
679
680 SelectObject(hdc, hOldPen);
681 DeleteObject(hPen);
682}
